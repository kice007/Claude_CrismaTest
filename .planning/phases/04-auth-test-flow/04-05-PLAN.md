// Read answers from sessionStorage; if missing, use fallback score 78
const storedAnswers = sessionStorage.getItem('crismatest_answers')
const answerCount = storedAnswers ? Object.keys(JSON.parse(storedAnswers)).length : 0
// Deterministic score: 60 + (answerCount * 3), clamped to 40–95
const finalScore = Math.min(95, Math.max(40, 60 + answerCount * 3))

const [finalScore, setFinalScore] = useState(0)
const [subScores, setSubScores] = useState({ logic: 0, communication: 0, jobSkill: 0, trust: 0, video: 0 })
const [grade, setGrade] = useState<'excellent' | 'good' | 'average' | 'below'>('good')
const [showConfetti, setShowConfetti] = useState(false)

useEffect(() => {
  // Read from sessionStorage (browser-only — must be in useEffect)

  const answerCount = storedAnswers ? Object.keys(JSON.parse(storedAnswers) as Record<string, unknown>).length : 0
  const score = Math.min(95, Math.max(40, 60 + answerCount * 3))
  setFinalScore(score)
  setSubScores({
    logic: Math.min(100, Math.round(score * 0.90)),

    jobSkill: Math.min(100, Math.round(score * 0.95)),
    trust: Math.min(100, Math.round(score * 0.85)),

  const g = score >= 85 ? 'excellent' : score >= 70 ? 'good' : score >= 55 ? 'average' : 'below'
