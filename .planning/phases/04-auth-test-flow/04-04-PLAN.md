**Permission request (called by "Check Permissions" button, NOT on mount):**
```typescript
const requestPermissions = async () => {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true })
    streamRef.current = stream

    // Assign stream to video element
    if (videoRef.current) {
      videoRef.current.srcObject = stream
    }

    // Web Audio API for mic level — created inside success callback (avoids autoplay block)
    const audioCtx = new AudioContext()
    const source = audioCtx.createMediaStreamSource(stream)
    const analyser = audioCtx.createAnalyser()
    analyser.fftSize = 256
    source.connect(analyser)
    analyserRef.current = analyser

    // requestAnimationFrame loop for mic bars
    const dataArray = new Uint8Array(analyser.frequencyBinCount)
    const tick = () => {
      analyser.getByteFrequencyData(dataArray)
      const avg = dataArray.reduce((a, b) => a + b, 0) / dataArray.length
      setMicLevel(avg / 255)
      animFrameRef.current = requestAnimationFrame(tick)
    }
    animFrameRef.current = requestAnimationFrame(tick)

    setPermissionGranted(true)
    // Auto-check camera and mic boxes
    setChecks(prev => ({ ...prev, camera: true, mic: true }))
  } catch {
    setPermissionDenied(true)
    toast.error(t('test_check_permission_denied'))
  }
}
```

**Cleanup on unmount:**
```typescript
useEffect(() => {
  return () => {
    cancelAnimationFrame(animFrameRef.current)
    streamRef.current?.getTracks().forEach(t => t.stop())

}, [])
```
