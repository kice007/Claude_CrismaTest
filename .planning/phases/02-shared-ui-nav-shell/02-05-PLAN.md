    Create src/components/nav/NavShell.tsx:

    ```tsx
    "use client";
    import { useEffect, useState } from "react";
    import { cn } from "@/lib/utils";
    import { NavDesktop } from "./NavDesktop";
    import { NavMobile } from "./NavMobile";

    export function NavShell() {
      const [scrolled, setScrolled] = useState(false);

      useEffect(() => {
        // Plain function inside useEffect — do NOT wrap in useCallback (React Compiler active)
        const handleScroll = () => setScrolled(window.scrollY > 8);
        window.addEventListener("scroll", handleScroll, { passive: true });
        return () => window.removeEventListener("scroll", handleScroll);
      }, []);

      return (
        <header
          role="banner"
          className={cn(
            "fixed top-0 left-0 right-0 z-50 transition-all duration-300",
            scrolled
              ? "bg-white/80 backdrop-blur-md shadow-[var(--shadow-card)]"
              : "bg-white/95 backdrop-blur-sm"
          )}
        >
          <NavDesktop />
          <NavMobile />
        </header>
      );
    }
    ```

    IMPORTANT implementation notes:
    - LogoMark SVG is duplicated in both NavDesktop and NavMobile because they render at different breakpoints. Alternatively, extract to src/components/nav/LogoMark.tsx and import in both — but duplication is acceptable for this size.
    - 'use client' is required on all three nav components: NavShell uses useEffect+useState, NavDesktop+NavMobile use useTranslation (client-only hook)
    - Do NOT use useCallback on scroll handler — React Compiler handles memoization
    - NavDesktop uses `hidden md:flex`, NavMobile uses `flex md:hidden` — they never render simultaneously
    - The hamburger button must set aria-label to the translated "open menu" string (via t("nav_open_menu"))
    - All nav links must go through useTranslation() — add these keys to /locales/en.json and /locales/fr.json:

    import { Inter, JetBrains_Mono } from "next/font/google";
    import { MotionConfig } from "motion/react";
    import { I18nProvider } from "@/components/I18nProvider";
    import { NavShell } from "@/components/nav/NavShell";
    import { Toaster } from "@/components/ui/sonner";

      subsets: ["latin"],
      variable: "--font-jetbrains-mono",
      display: "swap",
    });

    export const metadata: Metadata = {
      title: "CrismaTest",
      description: "AI-powered talent assessment platform",
      openGraph: {
        locale: "en_US",
        alternateLocale: ["fr_FR"],
      },
    };

    export default function RootLayout({
      children,
    }: Readonly<{
      children: React.ReactNode;
    }>) {
      return (
        <html lang="en" className={`${inter.variable} ${jetbrainsMono.variable}`}>
          <body className="antialiased overflow-x-hidden">
            <MotionConfig reducedMotion="user">
              <I18nProvider>
                <NavShell />
                <main className="pt-16 min-w-0">

                </main>
